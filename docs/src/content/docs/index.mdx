const quickstartSnippet = `import { Agent, run, tool } from '@openai/agents';
import { z } from 'zod';

type Weather = { city: string };

const getWeather = tool({
name: 'get_weather',
description: 'Get the weather for a given city',
parameters: z.object({ city: z.string() }),
execute: async ({ city }: Weather) => \`The weather in ${city} is sunny\`,
});

const assistant = new Agent({
name: 'Concierge',
instructions: 'Help builders explore the Agents SDK.',
tools: [getWeather],
});

const result = await run(
assistant,
'Share a launch checklist for my next agentic feature.',
);

console.log(result.finalOutput);
`;
